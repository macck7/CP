Brackets matching:
-------------------------
#include<bits/stdc++.h>
using namespace std;

unordered_map<char,int> symbols={{'(',-1},{'[',-2},{'{',-3},{')',1},{']',2},{'}',3}};

string isBalanced(string s){
	stack<char>st;
	for(char c : s){
		if(symbols[c]<0) st.push(c);
		else{
			if(st.empty()) return "NO";
			char top=st.top();
			st.pop();
			if(symbols[top]+ symbols[c] !=0) return "NO";
		}

	}if(st.empty()) return "YES";
	return "NO";
	
}

int main(){
	int n;
	cin >>n;
	while(n--){
		string s;
		cin>>s;
		cout<< isBalanced(s)<<endl;

		}

}
