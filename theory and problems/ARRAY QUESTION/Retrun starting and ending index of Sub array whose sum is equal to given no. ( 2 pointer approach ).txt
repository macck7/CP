#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,s;
	cin>>n >>s;
	int a[n];

	for(int i=0 ; i<n ; ++i){
		cin>> a[i];
	}
	
	int i=0,j=0 ,st=-1 ,ed = -1 , sum =0;

	while( j <n  and sum + a[j] <= s){
		sum += a[j];
		j += 1;
	}

	if( sum == s){
		cout<< i+1 <<" "<< j;  //j is not inclusive
		return 0;
	}
	
	while(j<n){
		 sum +=a[j];

		 while(sum> s){
		 	sum -=a[i];
		 	i++;
		 }

		 if( sum == s){
		 	st= i+1;   //i & j both are inclusive  so 1base return krna hai
		 	ed= j+1;
		 }
		 j++;
	}
	cout<< st<< " "<<ed ;
}
